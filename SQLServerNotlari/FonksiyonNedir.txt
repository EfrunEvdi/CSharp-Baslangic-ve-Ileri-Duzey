Sales.SalesOrderHeader SSOH
inner join Sales.SalesOrderDetail SSOD
on SSOH.SalesOrderID = SSOD.SalesOrderID
inner join Person.Person PP
on SSOH.SalesPersonID=PP.BusinessEntityID
where
SSOH.OrderDate = '2011-05-31'
